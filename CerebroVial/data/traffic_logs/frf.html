# Algorithm 1: Vehicular Flow Prediction Logic (Phase 2)
# Input: StreamDataEdge (JSON: Count, Speed, Crossing_ID)
# Output: Prediction_Vector (Future Flow for Phase 3)
# Resource: GRUModel (Pre-trained)

Start
    Load GRUModel into memory
    While System_Active do:
        Current_Data = ReadMessageQueue(StreamDataEdge)

        If Current_Data is NOT Null:
            Data_Norm = Normalize(Current_Data)
            Time_Window = GenerateSequence(Data_Norm, t-n)

            # Inference: Predict flow for the next t+1..t+k
            Future_Flow = GRUModel.Predict(Time_Window)

            # Package data for the Management Module
            Prediction_Vector = {Crossing_ID, Future_Flow, Timestamp}

            Send(Prediction_Vector) -> Traffic_Light_Management_Module
            Persist(Current_Data, Future_Flow) in Database
        End If
    End While
End
